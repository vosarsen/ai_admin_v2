Сильная дисторция

JavaScript:

const element = document.querySelector('#manual-1');
const glass = new GlassSurface(element, {
    borderRadius: 24,
    brightness: 60,
    opacity: 0.85,
    displace: 10
});

// Обновление параметров
glass.updateOptions({ brightness: 70 });

// Удаление эффекта
glass.destroy();
